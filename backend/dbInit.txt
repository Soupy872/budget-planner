CREATE TABLE users (
    ID SERIAL PRIMARY KEY,
    name VARCHAR(30),
    email VARCHAR(30)
);

CREATE TABLE income (
    ID SERIAL PRIMARY KEY,
    userid INTEGER,
    amount BIGINT,
    frequency VARCHAR(30)
);

CREATE TABLE expense (
    ID SERIAL PRIMARY KEY,
    userid INTEGER,
    amount BIGINT,
    frequency VARCHAR(30),
    categoryid INTEGER
);
